


0.положить папку ST7735 в папку с проектом

1.положить в папку inc проета файл main.h

2.скопировать 4 функции в main.с

void spi2_8b_init(void){//переходим в 8 бит SPI
 hspi1.Init.DataSize = SPI_DATASIZE_8BIT;
 HAL_SPI_Init(&hspi1);

}
void spi2_16b_init(void){//переходим в 16 бит SPI
 hspi1.Init.DataSize = SPI_DATASIZE_16BIT;
 HAL_SPI_Init(&hspi1);
}	
	
void lcd7735_senddata(unsigned char data) {//передаем в режиме 8 бит
  HAL_SPI_Transmit(&hspi1, &data,1,0x1);
}
void lcd7735_send16bData(uint8_t msb,uint8_t lsb) {//передаем в режиме 16 бит
	uint8_t masData[]={lsb,msb};
	HAL_SPI_Transmit(&hspi1,masData,1,0x1);
}

3.подключить в main.c следующие инклюды
#include "main.h"
#include "func.h"
4.В настройках проекта добавить путь к библиотеке ..\ST7735\inc
и подключить к проекту func.c

5.Если ,что то не понятно смотрим готовый проект

В данном проекте ножки дисплея подключены следующим образом
 LCD_RST GPIOA, GPIO_PIN_3
 LCD_DC  GPIOA, GPIO_PIN_4//a0//
 SCK     GPIOA, GPIO_PIN_5
 CS      GPIOA, GPIO_PIN_6
 SDA     GPIOA, GPIO_PIN_7
при изменении ног в проекте таже отредатировать main.h,при изменении номара SPI также изменить функции из 2-го пункта
